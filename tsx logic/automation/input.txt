    frame % 8,
    [0, 3, 4, 7],

    frame > 18 && frame < 24

        height="1080"

          <line
            key={i}
            x1={i * 70 - 200}
            y1={0}
            x2={i * 70 + 400}
            y2={1080}
            stroke="#ff0000"
            strokeWidth="1.5"
          />
        ))}
      </svg>

      {/* Shape layer — large tilted rectangle */}

        style={{
          position: 'absolute',
          width: 700,
          height: 700,

          borderRadius: 8,
          top: '15%',
          left: '62%',
