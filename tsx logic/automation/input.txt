import React from 'react';

import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Scene21_PayCut',
  durationInSeconds: 14,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Scene21_PayCut: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Camera slow zoom
  const camScale = interpolate(frame, [0, durationInFrames], [1.0, 1.06], { extrapolateRight: 'clamp' });

  // Paycheck slide-in
  const checkY = sp(10) * 1;
  const checkRot = interpolate(sp(10), [0, 1], [-8, -3]);

  // Scissor cut
  const scissorX = interpolate(frame, [60, 180], [-400, 1400], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.inOut(Easing.cubic) });
  const cutReveal = interpolate(frame, [120, 200], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const checkSplit = interpolate(frame, [180, 230], [0, 80], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.out(Easing.cubic) });

  // Number ticker (paycheck value going down)
  const tickerVal = interpolate(frame, [40, 200], [4250, 4071], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const displayVal = Math.floor(tickerVal);

  // Red bar drop
  const barDrop = sp(150) * 1;

  // Final caption pop
  const captionOp = interpolate(frame, [240, 280], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const captionScale = sp(240, { damping: 10, stiffness: 140 });

  return (
    <AbsoluteFill style={{ backgroundColor: '#A1BC98', overflow: 'hidden', opacity: exitOp }}>
      
      {/* Gradient overlay */}
      <AbsoluteFill style={{
        background: 'radial-gradient(circle at 30% 20%, rgba(238,238,238,0.18) 0%, transparent 55%), radial-gradient(circle at 75% 85%, rgba(255,87,34,0.12) 0%, transparent 60%)',
      }} />

      {/* Animated shape layers */}
      <div style={{
        position: 'absolute', width: 900, height: 900, borderRadius: '50%',
        border: '2px solid rgba(238,238,238,0.06)',
        top: -300, right: -200,
        transform: `rotate(${frame * 0.15}deg) scale(${1 + Math.sin(frame * 0.02) * 0.04})`,
      }} />
      <div style={{
        position: 'absolute', width: 600, height: 600,
        border: '2px solid rgba(255,87,34,0.08)',
        bottom: -200, left: -150,
        transform: `rotate(${-frame * 0.12}deg)`,
      }} />
      <div style={{
        position: 'absolute', width: 400, height: 400,
        background: 'rgba(238,238,238,0.04)',
        top: '40%', left: '5%',
        transform: `rotate(${frame * 0.08}deg) skew(15deg)`,
      }} />

      {/* Grid texture */}
      <svg width="1920" height="1080" style={{ position: 'absolute', opacity: 0.07 }}>
        <defs>
          <pattern id="grid21" width="80" height="80" patternUnits="userSpaceOnUse">
            <path d="M 80 0 L 0 0 0 80" fill="none" stroke="#EEEEEE" strokeWidth="1" />
          </pattern>
        </defs>
        <rect width="1920" height="1080" fill="url(#grid21)" />
      </svg>

      {/* Floating particles */}
      {Array.from({ length: 25 }).map((_, i) => {
        const px = seededRandom(i + 1) * 1920;
        const py = seededRandom(i + 50) * 1080;
        const ps = seededRandom(i + 100) * 4 + 2;
        const drift = Math.sin(frame * 0.02 + i) * 30;
        return (
          <div key={i} style={{
            position: 'absolute',
            left: px + drift, top: py + Math.cos(frame * 0.015 + i) * 20,
            width: ps, height: ps, borderRadius: '50%',
            background: i % 3 === 0 ? '#FF5722' : '#EEEEEE',
            opacity: 0.35,
          }} />
        );
      })}

      {/* Main scene container with camera */}
      <AbsoluteFill style={{ transform: `scale(${camScale})`, transformOrigin: 'center center' }}>

        {/* Top label */}
        <div style={{
          position: 'absolute', top: 90, left: '50%',
          transform: `translateX(-50%) translateY(${(1 - sp(0)) * -40}px)`,
          opacity: sp(0),
          fontFamily: 'JetBrains Mono, monospace',
          fontSize: 32, color: '#EEEEEE',
          letterSpacing: 6, fontWeight: 600,
          padding: '12px 32px',
          border: '2px solid #FF5722',
          background: 'rgba(28,28,28,0.35)',
        }}>
          THE INVISIBLE PAY CUT
        </div>

        {/* PAYCHECK CARD */}
        <div style={{
          position: 'absolute',
          top: '28%', left: '50%',

          transform: `translateX(calc(-50% - ${checkSplit}px)) translateY(${(1 - checkY) * 80}px) rotate(${checkRot}deg)`,

            fontFamily: 'Nexa Bold, sans-serif',

            {displayVal.toLocaleString()}
          </div>
          <div style={{
            fontFamily: 'Inter, sans-serif',
            fontSize: 32, color: '#1C1C1C', opacity: 0.7,
            marginTop: 10,
          }}>
            Same hours. Same role. Same number.
          </div>

          {/* Cut line visual */}
          {frame > 100 && (
            <div style={{
              position: 'absolute',
              top: 0, left: '50%',
              width: 3, height: '100%',
              background: '#FF5722',
              opacity: interpolate(frame, [100, 180], [0, 1], { extrapolateRight: 'clamp' }),
              boxShadow: '0 0 20px #FF5722',
            }} />
          )}
        </div>

        {/* RIGHT HALF (split piece) */}
        {frame > 180 && (
          <div style={{
            position: 'absolute',
            top: '28%', left: '50%',
            width: 450, height: 420,
            transform: `translateX(${checkSplit}px) rotate(${checkRot + 5}deg)`,

        {frame >= 60 && frame <= 180 && (
