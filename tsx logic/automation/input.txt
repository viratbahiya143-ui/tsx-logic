import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Ch3Scene1',
  durationInSeconds: 9,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Ch3Scene1: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Cracked/broken clock reveal
  const clockScale = sp(0, { damping: 10, stiffness: 80 });
  const crackReveal = interpolate(frame, [20, 60], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.out(Easing.cubic) });
  const textDrop = sp(45);
  const subDrop = sp(65);
  const redFlash = interpolate(frame, [30, 45, 60], [0, 0.4, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Particle drift
  const particles = Array.from({ length: 18 }, (_, i) => ({
    x: seededRandom(i * 7) * 1920,
    y: seededRandom(i * 13) * 1080,
    size: seededRandom(i * 3) * 6 + 2,
    speed: seededRandom(i * 17) * 0.4 + 0.1,
    drift: Math.sin(frame * seededRandom(i * 5) * 0.03 + i) * 30,
  }));

  return (
    <AbsoluteFill style={{ backgroundColor: '#1C1C1C', overflow: 'hidden', opacity: exitOp }}>

      {/* Deep red-black gradient bg */}
      <AbsoluteFill>
        <div style={{
          position: 'absolute', inset: 0,
          background: 'radial-gradient(ellipse at 30% 60%, #2a0a0a 0%, #1C1C1C 55%, #0d0d0d 100%)',
        }} />
      </AbsoluteFill>

      {/* Red flash overlay on impact */}
      <AbsoluteFill>
        <div style={{
          position: 'absolute', inset: 0,
          backgroundColor: '#ff0000',
          opacity: redFlash,
        }} />
      </AbsoluteFill>

      {/* Shape layer — large drifting triangle */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ position: 'absolute', inset: 0 }}>
          <polygon
            points="960,80 1400,900 520,900"
            fill="none"
            stroke="#ff0000"
            strokeWidth="1"
            opacity={0.05}
            transform={`rotate(${frame * 0.03}, 960, 540)`}
          />
          <polygon
            points="960,200 1250,780 670,780"
            fill="none"
            stroke="#e9d700"
            strokeWidth="0.5"
            opacity={0.04}
            transform={`rotate(${-frame * 0.02}, 960, 540)`}
          />
          {/* Grid lines texture */}
          {Array.from({ length: 12 }, (_, i) => (
            <line
              key={i}
              x1={i * 175} y1="0" x2={i * 175} y2="1080"
              stroke="#F5F5F5" strokeWidth="0.5" opacity={0.03}
            />
          ))}
          {Array.from({ length: 7 }, (_, i) => (
            <line
              key={i}
              x1="0" y1={i * 180} x2="1920" y2={i * 180}
              stroke="#F5F5F5" strokeWidth="0.5" opacity={0.03}
            />
          ))}
        </svg>
      </AbsoluteFill>

      {/* Floating particles */}
      <AbsoluteFill>
        <svg width="1920" height="1080" style={{ position: 'absolute' }}>
          {particles.map((p, i) => (
            <circle
              key={i}
              cx={p.x + p.drift}
              cy={(p.y - frame * p.speed * 20) % 1080}
              r={p.size}
              fill={i % 3 === 0 ? '#ff0000' : i % 3 === 1 ? '#e9d700' : '#F5F5F5'}
              opacity={0.12}
            />
          ))}
        </svg>
      </AbsoluteFill>

      {/* Broken clock SVG — center */}
      <AbsoluteFill style={{ display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
        <svg
          width="260" height="260"
          style={{
            transform: `scale(${clockScale}) translateX(-380px) translateY(20px)`,
            filter: 'drop-shadow(0 0 30px rgba(255,0,0,0.4))',

        >
          {/* Clock face */}

          <circle cx="130" cy="130" r="108" fill="none" stroke="#9E9E9E" strokeWidth="1" opacity={0.3} />

          marginBottom: 12,

            fontFamily: "'Noto Sans Devanagari', sans-serif",
            fontSize: 130,
            fontWeight: 900,

            letterSpacing: '-2px',

            खाली शाम
