import React from 'react';
import { useCurrentFrame, useVideoConfig, interpolate, spring, AbsoluteFill, Easing } from 'remotion';

export const compositionConfig = {
  id: 'Chapter3Scene1_SonekiTijori',
  durationInSeconds: 10,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Chapter3Scene1_SonekiTijori: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  // Camera zoom in
  const cameraZoom = interpolate(frame, [0, durationInFrames], [1, 1.08], {
    extrapolateRight: 'clamp',
  });

  // Gradient shift
  const gradAngle = interpolate(frame, [0, durationInFrames], [135, 165], {
    extrapolateRight: 'clamp',
  });

  // Vault door opening
  const vaultDoorLeft = interpolate(frame, [30, 70], [0, -420], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });
  const vaultDoorRight = interpolate(frame, [30, 70], [0, 420], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  // Golden glow from vault
  const glowOpacity = interpolate(frame, [40, 80], [0, 0.7], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });
  const glowScale = interpolate(frame, [40, 80], [0.3, 1.6], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  // Text reveal
  const titleY = interpolate(frame, [0, 1], [0, 0], { extrapolateRight: 'clamp' });
  const titleScale = sp(55, { damping: 10, stiffness: 100 });
  const subtitleOp = sp(75);

  // Chain/web overlay
  const chainOp = interpolate(frame, [80, 120], [0, 0.35], {
    extrapolateLeft: 'clamp',
    extrapolateRight: 'clamp',
  });

  // Particles
  const particles = Array.from({ length: 30 }, (_, i) => ({
    x: seededRandom(i * 7 + 1) * 1920,
    y: seededRandom(i * 7 + 2) * 1080,
    size: seededRandom(i * 7 + 3) * 5 + 2,
    speed: seededRandom(i * 7 + 4) * 0.6 + 0.2,
    opacity: seededRandom(i * 7 + 5) * 0.35 + 0.08,
    color: i % 3 === 0 ? '#FFC107' : i % 3 === 1 ? '#e9d700' : '#F5F5F5',
  }));

  // Shape layers
  const shapes = [
    { type: 'circle', x: 200, y: 180, size: 340, rot: frame * 0.12, op: 0.04 },
    { type: 'rect', x: 1500, y: 800, size: 260, rot: -frame * 0.08, op: 0.05 },
    { type: 'triangle', x: 1700, y: 200, size: 200, rot: frame * 0.15, op: 0.035 },
    { type: 'circle', x: 900, y: 900, size: 400, rot: -frame * 0.06, op: 0.03 },
  ];

  // Screen shake on vault open
  const shakeX = frame > 30 && frame < 50 ? Math.sin(frame * 2.5) * (4 - (frame - 30) * 0.2) : 0;
  const shakeY = frame > 30 && frame < 50 ? Math.cos(frame * 3.1) * (3 - (frame - 30) * 0.15) : 0;

  return (
    <AbsoluteFill
      style={{
        overflow: 'hidden',
        opacity: exitOp,
      }}
    >
      {/* Scaled wrapper for camera */}
      <div
        style={{
          width: '100%',
          height: '100%',
          transform: `scale(${cameraZoom}) translate(${shakeX}px, ${shakeY}px)`,
          transformOrigin: 'center center',
          position: 'absolute',
          top: 0,
          left: 0,
        }}
      >
        {/* Dynamic gradient background */}
        <AbsoluteFill
          style={{
            background: `linear-gradient(${gradAngle}deg, #0a0a0a 0%, #1C1C1C 35%, #2D2D2D 70%, #1a1205 100%)`,
          }}
        />

        {/* Animated shape layers */}
        {shapes.map((s, i) => (
          <div
            key={i}
            style={{
              position: 'absolute',
              left: s.x,
              top: s.y,
              width: s.size,
              height: s.size,
              opacity: s.op,
              transform: `rotate(${s.rot}deg) scale(${1 + Math.sin(frame * 0.02 + i) * 0.08})`,
              border:
                s.type === 'circle'
                  ? '2px solid #FFC107'
                  : s.type === 'rect'
                  ? '2px solid #e9d700'
                  : '2px solid #F5F5F5',
              borderRadius: s.type === 'circle' ? '50%' : s.type === 'rect' ? '12px' : '0',
              clipPath: s.type === 'triangle' ? 'polygon(50% 0%, 0% 100%, 100% 100%)' : undefined,
            }}
          />
        ))}

        {/* Motion texture — diagonal gold lines */}
        <AbsoluteFill style={{ opacity: 0.04 }}>
          {Array.from({ length: 18 }, (_, i) => (
            <div
              key={i}
              style={{
                position: 'absolute',
                left: -200 + i * 140 + Math.sin(frame * 0.015 + i) * 30,
                top: 0,
                width: 1,
                height: 2200,
                background: '#FFC107',
                transform: `rotate(25deg)`,
              }}
            />
          ))}
        </AbsoluteFill>

        {/* Vault structure — dark rectangle */}
        <div
          style={{
            position: 'absolute',
            left: '50%',
            top: '50%',
            transform: 'translate(-50%, -50%)',
            width: 600,
            height: 500,
            background: 'linear-gradient(180deg, #2D2D2D 0%, #1C1C1C 100%)',
            borderRadius: 24,
            border: '4px solid #555',
            boxShadow: '0 0 80px rgba(0,0,0,0.8)',
            overflow: 'hidden',
          }}
        >
          {/* Golden glow from inside */}
          <div
            style={{
              position: 'absolute',
              left: '50%',
              top: '50%',
              transform: `translate(-50%, -50%) scale(${glowScale})`,
              width: 300,
              height: 300,
              borderRadius: '50%',
              background: 'radial-gradient(circle, rgba(255,193,7,0.6) 0%, rgba(233,215,0,0.2) 40%, transparent 70%)',
              opacity: glowOpacity,
            }}
          />

          {/* Gold bars inside vault */}
          {glowOpacity > 0.2 && (
            <div style={{ position: 'absolute', left: '50%', top: '55%', transform: 'translate(-50%, -50%)' }}>
              {/* Stack of gold bars */}
              {[0, 1, 2].map((row) =>
                Array.from({ length: 3 - row }, (_, col) => (
                  <div
                    key={`${row}-${col}`}
                    style={{
                      position: 'absolute',
                      left: -60 + col * 55 + row * 25,
                      top: -row * 28,
                      width: 50,
                      height: 22,
                      background: `linear-gradient(180deg, #FFD54F, #FFC107, #FF8F00)`,
                      borderRadius: 3,
                      border: '1px solid #FFB300',
                      opacity: interpolate(frame, [50 + row * 10, 65 + row * 10], [0, 1], {
                        extrapolateLeft: 'clamp',
                        extrapolateRight: 'clamp',
                      }),
                      boxShadow: '0 2px 8px rgba(255,193,7,0.4)',
                    }}
                  />
                ))
              )}
            </div>
          )}

          {/* Vault door LEFT */}
          <div
            style={{
              position: 'absolute',
              left: 0,
              top: 0,
              width: '50%',
              height: '100%',
              background: 'linear-gradient(90deg, #3a3a3a, #4a4a4a)',
              transform: `translateX(${vaultDoorLeft}px)`,
              borderRight: '3px solid #555',
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'flex-end',
              paddingRight: 20,
            }}
          >
            {/* Door handle */}
            <div
              style={{
                width: 40,
                height: 40,
                borderRadius: '50%',
                border: '4px solid #888',
                background: '#2D2D2D',
              }}
            />
          </div>

          {/* Vault door RIGHT */}
          <div
            style={{
              position: 'absolute',
              right: 0,
              top: 0,
              width: '50%',
              height: '100%',
              background: 'linear-gradient(270deg, #3a3a3a, #4a4a4a)',
              transform: `translateX(${vaultDoorRight}px)`,
              borderLeft: '3px solid #555',
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'flex-start',
              paddingLeft: 20,
            }}
          >
            <div
              style={{
                width: 40,
                height: 40,
                borderRadius: '50%',
                border: '4px solid #888',
                background: '#2D2D2D',
              }}
            />
          </div>
        </div>

        {/* Chain/web overlay — golden web pattern */}
        <AbsoluteFill style={{ opacity: chainOp }}>
          <svg width="1920" height="1080" viewBox="0 0 1920 1080" style={{ position: 'absolute' }}>
            {Array.from({ length: 12 }, (_, i) => {
              const cx = 960;
              const cy = 540;
              const angle = (i * 30 * Math.PI) / 180;
              const r = 500;
              return (
                <line
                  key={i}
                  x1={cx}
                  y1={cy}
                  x2={cx + Math.cos(angle) * r}
                  y2={cy + Math.sin(angle) * r}
                  stroke="#FFC107"
                  strokeWidth="1"
                  opacity="0.3"
                />
              );
            })}
            {[150, 300, 450].map((r, i) => (
              <circle key={i} cx="960" cy="540" r={r} fill="none" stroke="#FFC107" strokeWidth="0.8" opacity="0.2" />
            ))}
          </svg>
        </AbsoluteFill>

        {/* Floating particles */}
        {particles.map((p, i) => (
          <div
            key={i}
            style={{
              position: 'absolute',
              left: p.x,
              top: (p.y + frame * p.speed * 3) % 1120 - 40,

              height: p.size,
              borderRadius: '50%',

        {/* Title text — top area */}

          style={{
            position: 'absolute',
            top: 60,
            left: 0,
            width: '100%',
            display: 'flex',
            flexDirection: 'column',
            alignItems: 'center',
            transform: `scale(${titleScale})`,
          }}
        >
          <div
            style={{
              fontFamily: 'Yantra One',
              fontSize: 110,
              color: '#FFC107',

              lineHeight: 1.1,
              textAlign: 'center',
            }}
          >
            सोने की तिजोरी
          </div>
        </div>

        {/* Subtitle */}
        <div
          style={{
            position: 'absolute',
            bottom: 140,
            left: 0,
            width: '100%',
            display: 'flex',
            justifyContent: 'center',
            opacity: subtitleOp,
          }}
        >
          <div
            style={{
              fontFamily: 'Noto Sans Devanagari',
              fontSize: 48,
              color: '#F5F5F5',
              background: 'rgba(28,28,28,0.7)',
              padding: '14px 50px',
              borderRadius: 14,
              border: '1px solid rgba(255,193,7,0.3)',
              textShadow: '0 2px 10px rgba(0,0,0,0.6)',
            }}
          >
            जाल और भी गहरा है...
          </div>
        </div>

        {/* Stick figure with golden glasses — bottom right area */}
        <div
          style={{
            position: 'absolute',
            right: 160,
            bottom: 140,
            transform: `translateY(${Math.sin(frame * 0.05) * 2}px) scale(${sp(45)})`,
          }}
        >
          {/* Golden glow behind */}
          <div
            style={{
              position: 'absolute',
              left: '50%',
              top: '40%',
              transform: 'translate(-50%, -50%)',
              width: 90,

          />
          <svg width="80" height="130" viewBox="0 0 80 130">

            <line x1="54" y1="20" x2="58" y2="18" stroke="#FFC107" strokeWidth="2" />
            {/* Eyes — concerned */}
            <circle cx="33" cy="19" r="2" fill="#F5F5F5" />
            <circle cx="47" cy="19" r="2" fill="#F5F5F5" />
            {/* Mouth — slight frown */}
            <path d="M34 28 Q40 25 46 28" fill="none" stroke="#F5F5F5" strokeWidth="1.5" />

            <line x1="40" y1="40" x2="40" y2="85" stroke="#F5F5F5" strokeWidth="3" />

              x1="40"
              y1="55"
              x2={15 + Math.sin(frame * 0.04) * 3}
              y2="70"
              stroke="#F5F5F5"
