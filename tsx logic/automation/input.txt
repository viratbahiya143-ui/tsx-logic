	AbsoluteFill,
	useCurrentFrame,
	useVideoConfig,
	interpolate,
	spring,
	Easing,

export const compositionConfig = {
	id: 'Chapter3Scene1',
	durationInSeconds: 12,
	fps: 30,
	width: 1920,
	height: 1080,
};

const seededRandom = (seed: number): number => {
	const x = Math.sin(seed * 9999) * 10000;
	return x - Math.floor(x);
};

const Chapter3Scene1: React.FC = () => {
	const frame = useCurrentFrame();
	const {fps, durationInFrames} = useVideoConfig();

	const sp = (delay: number, config?: object) =>
		spring({
			frame: Math.max(0, frame - delay),
			fps,
			config: {damping: 12, mass: 0.4, stiffness: 120, ...config},
		});

	const exitOp = interpolate(
		frame,
		[durationInFrames - 15, durationInFrames],
		[1, 0],
		{extrapolateLeft: 'clamp', extrapolateRight: 'clamp'}
	);

	// Camera zoom
	const cameraZoom = interpolate(frame, [0, durationInFrames], [1, 1.06], {
		extrapolateRight: 'clamp',
	});

	const gradAngle = interpolate(frame, [0, durationInFrames], [135, 165]);

	// ₹1,00,000 counter animation
	const counterProgress = interpolate(frame, [30, 90], [0, 100000], {
		extrapolateLeft: 'clamp',
		extrapolateRight: 'clamp',
		easing: Easing.out(Easing.cubic),
	});
	const displayAmount = Math.floor(counterProgress);
	const formattedAmount = displayAmount.toLocaleString('en-IN');

	// Coin SVGs falling
	const coins = Array.from({length: 18}, (_, i) => {
		const x = seededRandom(i * 7 + 1) * 1920;
		const delay = seededRandom(i * 7 + 2) * 80;
		const speed = 1.5 + seededRandom(i * 7 + 3) * 2.5;
		const size = 18 + seededRandom(i * 7 + 4) * 22;
		const yPos = interpolate(
			frame - delay,
			[0, durationInFrames],
			[-50, 1200],
			{extrapolateLeft: 'clamp', extrapolateRight: 'clamp'}
		);
		const rotation = (frame - delay) * speed * 2;
		const opacity = interpolate(frame - delay, [0, 20, durationInFrames - 30, durationInFrames], [0, 0.25, 0.25, 0], {
			extrapolateLeft: 'clamp',
			extrapolateRight: 'clamp',
		});
		return {x, yPos, size, rotation, opacity, id: i};
	});

	// Shape layers
	const shape1Rot = frame * 0.15;
	const shape2Rot = frame * -0.1;
	const shape3Scale = 1 + Math.sin(frame * 0.02) * 0.08;

	// Particles
	const particles = Array.from({length: 35}, (_, i) => {
		const px = seededRandom(i * 11 + 50) * 1920;
		const py = seededRandom(i * 11 + 51) * 1080;
		const pSize = 2 + seededRandom(i * 11 + 52) * 3;
		const pSpeed = 0.3 + seededRandom(i * 11 + 53) * 0.7;
		const pOpacity = 0.15 + seededRandom(i * 11 + 54) * 0.25;
		const yOffset = Math.sin((frame + i * 20) * 0.03 * pSpeed) * 30;
		const xOffset = Math.cos((frame + i * 15) * 0.02 * pSpeed) * 20;
		return {px: px + xOffset, py: py + yOffset, pSize, pOpacity, id: i};
	});

	// "₹1 लाख" hero text
	const heroScale = sp(20, {damping: 10, stiffness: 80});
	const heroY = interpolate(heroScale, [0, 1], [80, 0]);

	// Subtitle
	const subOp = sp(55);
	const subY = interpolate(subOp, [0, 1], [40, 0]);

	// "कुछ नहीं है" punch text
	const punchScale = sp(90, {damping: 8, stiffness: 150});
	const punchRotate = interpolate(punchScale, [0, 1], [-5, 0]);

	// Screen shake on counter hit
	const shakeIntensity = interpolate(frame, [88, 92, 100], [0, 6, 0], {
		extrapolateLeft: 'clamp',
		extrapolateRight: 'clamp',
	});
	const shakeX = Math.sin(frame * 15) * shakeIntensity;
	const shakeY = Math.cos(frame * 18) * shakeIntensity;

	// Question mark decoration
	const qScale = sp(40, {damping: 14});

	// Stick figure
	const figureEntry = sp(45);
	const figureX = interpolate(figureEntry, [0, 1], [200, 0]);
	const breathe = Math.sin(frame * 0.05) * 2;
	const headTilt = Math.sin(frame * 0.04) * 3;
	// Shrug animation
	const shoulderShrug = Math.sin(frame * 0.08) * 4;

	// Grid texture
	const gridOp = interpolate(frame, [0, 30], [0, 0.04], {
		extrapolateRight: 'clamp',
	});

	return (
		<AbsoluteFill
			style={{
				background: bg,
				overflow: 'hidden',
				opacity: exitOp,
			}}
		>
			<div
				style={{
					position: 'absolute',
					inset: 0,
					transform: `scale(${cameraZoom}) translate(${shakeX}px, ${shakeY}px)`,
				}}
			>
				{/* Grid texture */}
				<svg
					style={{
						position: 'absolute',
						inset: 0,
						width: '100%',
						height: '100%',
						opacity: gridOp,
					}}
				>
					{Array.from({length: 20}, (_, i) => (
						<line
							key={`h${i}`}
							x1="0"
							y1={i * 56}
							x2="1920"
							y2={i * 56}
							stroke="#FFC107"
							strokeWidth="0.5"
						/>
					))}
					{Array.from({length: 35}, (_, i) => (
						<line
							key={`v${i}`}
							x1={i * 56}
							y1="0"
							x2={i * 56}
							y2="1080"
							stroke="#FFC107"
							strokeWidth="0.5"
						/>
					))}
				</svg>

				{/* Shape layers */}
				<div
					style={{
						position: 'absolute',
						top: '10%',
						left: '5%',
						width: 400,
						height: 400,
						borderRadius: '50%',
						border: '2px solid rgba(233,215,0,0.05)',
						transform: `rotate(${shape1Rot}deg)`,
					}}
				/>
				<div
					style={{
						position: 'absolute',
						bottom: '15%',
						right: '8%',
						width: 300,
						height: 300,
						border: '2px solid rgba(255,193,7,0.04)',
						transform: `rotate(${shape2Rot}deg)`,
					}}
				/>
				<div
					style={{
						position: 'absolute',
						top: '50%',
						left: '50%',
						width: 500,
						height: 500,
						borderRadius: '50%',
						border: '1.5px solid rgba(245,245,245,0.03)',
						transform: `translate(-50%, -50%) scale(${shape3Scale})`,
					}}
				/>
				{/* Triangle shape */}
				<svg
					style={{
						position: 'absolute',
						top: '20%',
						right: '20%',
						opacity: 0.04,
						transform: `rotate(${frame * 0.12}deg)`,
					}}
					width="250"
					height="250"
					viewBox="0 0 250 250"
				>
					<polygon
						points="125,20 240,220 10,220"
						fill="none"
						stroke="#FFC107"
						strokeWidth="2"
					/>
				</svg>

				{/* Falling coins */}
				{coins.map((coin) => (
					<svg
						key={coin.id}
						style={{
							position: 'absolute',
							left: coin.x,
							top: coin.yPos,
							width: coin.size,
							height: coin.size,
							opacity: coin.opacity,
							transform: `rotate(${coin.rotation}deg)`,
						}}
						viewBox="0 0 40 40"
					>
						<circle cx="20" cy="20" r="18" fill="#FFC107" />
						<text
							x="20"
							y="26"
							textAnchor="middle"
							fontSize="18"
							fontWeight="bold"
							fill="#1C1C1C"
						>
							₹
						</text>
					</svg>
				))}

				{/* Particles */}
				{particles.map((p) => (
					<div
						key={p.id}
						style={{
							position: 'absolute',
							left: p.px,
							top: p.py,
							width: p.pSize,
							height: p.pSize,
							borderRadius: '50%',
							backgroundColor: '#FFC107',
							opacity: p.pOpacity,

				{/* Main content area */}

								lineHeight: 1,

							}}
						>
							₹{formattedAmount}
						</div>
					</div>

					{/* "एक लाख रूपी" */}
					<div
						style={{
							opacity: subOp,
