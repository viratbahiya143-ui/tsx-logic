  fps: 30,
  width: 1920,
  height: 1080,
};

  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);

    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',

  const camZoom = interpolate(frame, [0, durationInFrames], [1, 1.06], { extrapolateRight: 'clamp' });

  // Door closing animation
  const doorRotate = interpolate(frame, [60, 120], [-75, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp', easing: Easing.bezier(0.65, 0, 0.35, 1),
  });
  const doorShake = frame > 120 && frame < 140 ? Math.sin(frame * 1.5) * 4 : 0;

  // Date stamp
  const stampScale = sp(140, { damping: 8, stiffness: 200 });
  const stampRotate = interpolate(stampScale, [0, 1], [-15, -8]);

  // Lock icon
  const lockOp = interpolate(frame, [160, 190], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });
  const lockY = interpolate(frame, [160, 190], [30, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Headline reveal
  const headlineY = sp(15) * 1;
  const subOp = interpolate(frame, [40, 70], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  // Bottom badge
  const badgeOp = interpolate(frame, [200, 230], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  return (
    <AbsoluteFill style={{ backgroundColor: '#412D15', overflow: 'hidden', opacity: exitOp }}>
      {/* Gradient background */}
      <AbsoluteFill
        style={{
          background: 'radial-gradient(ellipse at 30% 20%, rgba(250,185,91,0.18) 0%, rgba(65,45,21,0) 55%), radial-gradient(ellipse at 80% 90%, rgba(232,237,242,0.06) 0%, rgba(65,45,21,0) 60%)',
        }}
      />

      {/* Animated shape layers */}
      <AbsoluteFill>
        {[...Array(5)].map((_, i) => {
          const seed = i + 1;
          const baseX = seededRandom(seed) * 1920;
          const baseY = seededRandom(seed * 2) * 1080;
          const drift = Math.sin(frame * 0.008 + i) * 40;
          const breath = 1 + Math.sin(frame * 0.02 + i) * 0.08;
          const rot = frame * 0.05 * (i % 2 === 0 ? 1 : -1);
          const size = 280 + seededRandom(seed * 3) * 200;
          return (
            <div
              key={`shape-${i}`}
              style={{
                position: 'absolute',
                left: baseX + drift,
                top: baseY,
                width: size,
                height: size,
                border: `2px solid rgba(232,237,242,${0.04 + (i % 3) * 0.015})`,
                borderRadius: i % 2 === 0 ? '50%' : '12px',
                transform: `scale(${breath}) rotate(${rot}deg)`,
                opacity: 0.6,
              }}
            />
          );
        })}
      </AbsoluteFill>

      {/* Motion texture - diagonal grid lines */}
      <AbsoluteFill style={{ opacity: 0.08 }}>
        <svg width="1920" height="1080" style={{ position: 'absolute' }}>
          <defs>
            <pattern id="grid21" x="0" y="0" width="80" height="80" patternUnits="userSpaceOnUse" patternTransform={`translate(${frame * 0.3} 0)`}>
              <path d="M 80 0 L 0 0 0 80" fill="none" stroke="#E8EDF2" strokeWidth="1" />
            </pattern>
          </defs>
          <rect width="1920" height="1080" fill="url(#grid21)" />
        </svg>
      </AbsoluteFill>

      {/* Floating particles */}
      <AbsoluteFill>
        {[...Array(28)].map((_, i) => {
          const seed = i + 10;
          const px = (seededRandom(seed) * 1920 + frame * (seededRandom(seed * 2) * 0.6 + 0.2)) % 1920;
          const py = (seededRandom(seed * 3) * 1080 + Math.sin(frame * 0.02 + i) * 30);
          const pSize = 2 + seededRandom(seed * 4) * 4;
          return (
            <div
              key={`p-${i}`}
              style={{
                position: 'absolute',
                left: px,
                top: py,
                width: pSize,
                height: pSize,
                borderRadius: '50%',
                background: i % 3 === 0 ? '#FAB95B' : '#E8EDF2',
                opacity: 0.35,
                boxShadow: i % 3 === 0 ? '0 0 8px #FAB95B' : 'none',
              }}
            />
          );
        })}
      </AbsoluteFill>

      {/* Main content with camera zoom */}
      <AbsoluteFill style={{ transform: `scale(${camZoom})`, transformOrigin: 'center' }}>
        {/* Top headline */}
        <div
          style={{
            position: 'absolute',
            top: 100,
            left: 0,
            right: 0,
            textAlign: 'center',
            transform: `translateY(${(1 - headlineY) * -40}px)`,
            opacity: headlineY,
          }}
        >
          <div
            style={{
              fontFamily: 'Inter, Poppins',
              fontSize: 51,
              fontWeight: 600,
              color: '#FAB95B',
              letterSpacing: 4,
              textTransform: 'uppercase',
              marginBottom: 18,

          >
            Born After The Cutoff

              borderRadius: 8,

              boxShadow: '0 0 60px rgba(0,0,0,0.5)',

          />

          <div
            style={{
              position: 'absolute',
              top: 6,
              left: 6,
              width: 'calc(100% - 12px)',
              height: 'calc(100% - 12px)',
              background: 'linear-gradient(135deg, #5a3e1f 0%, #2d1d0c 100%)',
              borderRight: '3px solid #FAB95B',
              transformOrigin: 'left center',
