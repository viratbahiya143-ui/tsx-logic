import React from 'react';

import { useCurrentFrame, useVideoConfig, interpolate, Easing, spring, AbsoluteFill } from 'remotion';

export const compositionConfig = {
  id: 'Scene01',
  durationInSeconds: 13,
  fps: 30,
  width: 1920,
  height: 1080,
};

const seededRandom = (seed: number): number => {
  const x = Math.sin(seed * 9999) * 10000;
  return x - Math.floor(x);
};

const Scene01: React.FC = () => {
  const frame = useCurrentFrame();
  const { fps, durationInFrames } = useVideoConfig();

  const sp = (delay: number, config?: object) =>
    spring({ frame: Math.max(0, frame - delay), fps, config: { damping: 12, mass: 0.4, stiffness: 120, ...config } });

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Question mark draw
  const qDraw = interpolate(frame, [10, 60], [0, 1], { extrapolateRight: 'clamp', easing: Easing.out(Easing.cubic) });
  const qScale = sp(5, { damping: 10, stiffness: 90 });

  // Heart pulse
  const heartPulse = 1 + Math.sin(frame * 0.15) * 0.08;
  const heartOp = interpolate(frame, [70, 100], [0, 1], { extrapolateRight: 'clamp' });

  // Text reveals
  const t1 = sp(90);
  const t2 = sp(140);

  // Camera zoom
  const camZoom = interpolate(frame, [0, durationInFrames], [1, 1.06], { easing: Easing.inOut(Easing.quad) });

  return (
    <AbsoluteFill style={{ backgroundColor: '#E1CCEC', overflow: 'hidden', opacity: exitOp }}>
      {/* Gradient overlay */}
      <AbsoluteFill style={{
        background: 'radial-gradient(circle at 50% 40%, rgba(246,36,64,0.08) 0%, transparent 60%)',
      }} />

      {/* Shape layers - drifting */}
      <div style={{
        position: 'absolute', width: 700, height: 700, borderRadius: '50%',
        border: '2px solid #000000', opacity: 0.05,
        top: -200, left: -200,
        transform: `rotate(${frame * 0.2}deg) scale(${1 + Math.sin(frame * 0.02) * 0.05})`,
      }} />
      <div style={{
        position: 'absolute', width: 500, height: 500,
        border: '2px solid #F62440', opacity: 0.06,
        bottom: -150, right: -100,
        transform: `rotate(${-frame * 0.15}deg)`,
      }} />

      {/* Grid texture */}
      <svg style={{ position: 'absolute', width: '100%', height: '100%', opacity: 0.07 }}>
        <defs>
          <pattern id="grid1" width="80" height="80" patternUnits="userSpaceOnUse">
            <path d="M 80 0 L 0 0 0 80" fill="none" stroke="#000000" strokeWidth="1" />
          </pattern>
        </defs>
        <rect width="100%" height="100%" fill="url(#grid1)" />
      </svg>

      {/* Floating particles */}
      {Array.from({ length: 18 }).map((_, i) => {
        const px = seededRandom(i + 1) * 1920;
        const py = seededRandom(i + 100) * 1080;
        const drift = Math.sin(frame * 0.02 + i) * 20;
        return (
          <div key={i} style={{
            position: 'absolute', left: px + drift, top: py,
            width: 6, height: 6, borderRadius: '50%',
            backgroundColor: '#F62440', opacity: 0.3,
          }} />
        );
      })}

      <AbsoluteFill style={{ transform: `scale(${camZoom})`, transformOrigin: 'center center' }}>
        {/* Giant Question Mark SVG */}
        <div style={{
          position: 'absolute', left: '50%', top: 180,
          transform: `translateX(-50%) scale(${qScale})`,
        }}>
          <svg width="280" height="380" viewBox="0 0 280 380">
            <path
              d="M 60 100 Q 60 30 140 30 Q 220 30 220 100 Q 220 160 160 190 Q 140 200 140 240"
              fill="none" stroke="#F62440" strokeWidth="22" strokeLinecap="round"
              strokeDasharray="700" strokeDashoffset={700 - qDraw * 700}
            />
            <circle cx="140" cy="310" r="18" fill="#000000"
              opacity={interpolate(frame, [55, 70], [0, 1], { extrapolateRight: 'clamp' })} />
          </svg>
        </div>

        {/* Heart icon */}
        <div style={{
          position: 'absolute', left: '50%', top: 580,
          transform: `translateX(-50%) scale(${heartPulse})`, opacity: heartOp,
        }}>
          <svg width="90" height="90" viewBox="0 0 100 100">
            <path d="M50 85 C 20 60, 10 35, 30 25 C 42 19, 50 30, 50 30 C 50 30, 58 19, 70 25 C 90 35, 80 60, 50 85 Z"
              fill="#F62440" />
          </svg>
        </div>

        {/* Headline text */}
        <div style={{
          position: 'absolute', left: 0, right: 0, top: 720, textAlign: 'center',
          opacity: t1, transform: `translateY(${(1 - t1) * 30}px)`,
        }}>
          <div style={{
            fontFamily: 'Yantra One, serif', fontSize: 119, color: '#000000',
            fontWeight: 700, letterSpacing: '-2px', lineHeight: 1,
          }}>
            एक सवाल
          </div>
        </div>

        <div style={{
          position: 'absolute', left: 0, right: 0, top: 870, textAlign: 'center',

        background: 'radial-gradient(ellipse at center, transparent 50%, rgba(0,0,0,0.25) 100%)',

        color: '#9E9E9E', opacity: 0.3,
      }}>
        ProBuffet HQ
      </div>
    </AbsoluteFill>
  );
};

export default Scene01;
