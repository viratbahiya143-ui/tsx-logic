  width: 1920,
  height: 1080,

  const exitOp = interpolate(frame, [durationInFrames - 15, durationInFrames], [1, 0], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
  });

  // Crack reveal across screen
  const crackProgress = interpolate(frame, [0, 40], [0, 1], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.out(Easing.cubic),
  });

  // Brain pulse
  const brainPulse = Math.sin(frame * 0.12) * 8;
  const brainScale = 1 + Math.sin(frame * 0.08) * 0.03;

  // Red invasion from top
  const redHeight = interpolate(frame, [20, 80], [0, 380], {
    extrapolateLeft: 'clamp', extrapolateRight: 'clamp',
    easing: Easing.inOut(Easing.cubic),

  const line1Op = interpolate(frame, [30, 50], [0, 1], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' });

  const shakeX = frame > 70 && frame < 78 ? (Math.sin(frame * 3.7) * 8) : 0;
  const shakeY = frame > 70 && frame < 78 ? (Math.sin(frame * 4.1) * 5) : 0;

          />

          {Array.from({ length: 12 }, (_, i) => (
